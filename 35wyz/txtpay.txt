    function addno(num) { 
    
        var ticketNoStr; var TicketNum;
        var numstart = parseInt(num,10); 
            while (true)
                { 
                    var TicketNo = document.getElementById("TicketNo" + numstart);
                    if (TicketNo != null)
                    {
                        if (numstart == parseInt(num, 10))//起始票号值必须输入
                        {
                            if (TicketNo.value == null || TicketNo.value == "") {
                                alert("请输入票号"); return;
                            }
                            TicketNo.value = Trim(TicketNo.value);
                            ticketNoStr = TicketNo.value;
                            try {
                                if (ticketNoStr.indexOf('-') > 0) {
                                    if (ticketNoStr.split('-')[1] != "") {
                                        TicketNum = parseInt(ticketNoStr.split('-')[1], 10);
                                        if (isNaN(TicketNum) || isNaN(ticketNoStr.split('-')[1]))
                                        { alert("票号输入有误"); return; }
                                    }
                                }
                                else {
                                    TicketNum = parseInt(ticketNoStr, 10);
                                    if (isNaN(TicketNum) || isNaN(ticketNoStr))
                                    { alert("票号输入有误"); return; }
                                }
 
                            }
                            catch (err) { alert("票号输入有误"); return; }
                        }
                        else {
                            if (!isNaN(TicketNum)) {
                                if (ticketNoStr.indexOf('-') > 0) {
                                    TicketNo.value = ticketNoStr.split('-')[0] + "-" + TicketNum;
                                }
                                else {
                                    TicketNo.value = TicketNum;
                                }
                            }
                        }
                    }
                    else 
                    {
                        break;
                    }
                    numstart++;
                    TicketNum++;
                }
            }
